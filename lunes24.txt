create or replace PROCEDURE    Ver_NombrePais 
(PARAM_COD_PAIS IN varchar, PARAM_NOMBRE_PAIS OUT VARCHAR ) AS
BEGIN    
      SELECT COUNTRY_NAME  INTO PARAM_NOMBRE_PAIS
      FROM HR.COUNTRIES    
      WHERE COUNTRY_ID = PARAM_COD_PAIS; 
END;
